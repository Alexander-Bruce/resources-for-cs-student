​	数据库连接：这段代码定义了一个名为 `DBConnection` 的 Java 类，用于管理与 SQL Server 数据库的连接。它包含两个主要方法：`getConnection` 和 `dbClose`。`getConnection` 方法加载 SQL Server JDBC 驱动，创建并返回一个数据库连接对象，连接到指定的数据库（`education`）并使用提供的用户名和密码进行身份验证。`dbClose` 方法则用于安全地关闭数据库连接、预处理语句和结果集，避免资源泄露。整体上，这个类为后续的数据库操作提供了基础设施。

```
package backend.database_jdbc;

import java.sql.*;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

public class DBConnection {

    public static Connection getConnection() throws Exception {
        Connection conn = null;

        Class.forName("com.microsoft.sqlserver.jdbc.SQLServerDriver");

        String url = "jdbc:sqlserver://localhost:1433;databaseName=education";
        conn = DriverManager.getConnection(url, "SA", "Heqinglin2021@"); // 替换为实际的用户名和密码

        return conn;
    }

    public static void dbClose(Connection conn, PreparedStatement ps, ResultSet rs) throws SQLException {
        if (rs != null) {
            rs.close();
        }
        if (ps != null) {
            ps.close();
        }
        if (conn != null) {
            conn.close();
        }
    }
}
```

​	数据库展示：这段 JSP 代码用于展示学生信息的网页。页面的标题为“学生信息”，包含一个表格，表格头部定义了六个列：学号、姓名、性别、专业班级、出生日期和电话号码。在脚本部分，首先创建数据库连接，执行 SQL 查询以获取 `student` 表中的所有记录。使用 `PreparedStatement` 来执行查询，并通过 `ResultSet` 遍历每一行数据。在遍历过程中，将每个字段的值提取并填充到表格的相应单元格中。对于出生日期和电话号码，如果它们为空，表格会显示“无”。最后，代码在 `finally` 块中关闭数据库连接，确保不产生资源泄露。整体上，该代码实现了从数据库动态加载学生信息并以表格形式展示在网页上。

public void insert() {
    Connection conn = null;

        ps = conn.prepareStatement(selectSQL);
        rs = ps.executeQuery();

        System.out.println("学号\t姓名\t性别\t专业班级\t出生日期\t电话号码");
        while (rs.next()) {

    } catch (SQLException e) {
        e.printStackTrace();
    } catch (Exception e) {
        throw new RuntimeException(e);
    } finally {
        // 关闭连接
        try {
            DBConnection.dbClose(conn, ps, rs);
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}

import com.alibaba.druid.pool.DruidDataSource;

import java.sql.Connection;
import java.sql.SQLException;

public class DruidDBConnection {
    // 创建 Druid 数据源实例
    private static DruidDataSource dataSource = new DruidDataSource();

    static {
        // 设置数据库连接信息
        dataSource.setUrl("jdbc:sqlserver://localhost:1433;databaseName=education"); // 数据库 URL
        dataSource.setUsername("SA"); // 数据库用户名
        dataSource.setPassword("Heqinglin2021@"); // 数据库密码
        dataSource.setDriverClassName("com.microsoft.sqlserver.jdbc.SQLServerDriver"); // JDBC 驱动类
        dataSource.setInitialSize(5); // 初始化连接数
        dataSource.setMinIdle(5); // 最小空闲连接数
        dataSource.setMaxActive(20); // 最大活动连接数
        dataSource.setMaxWait(60000); // 获取连接的最大等待时间（毫秒）
        dataSource.setTestOnBorrow(true); // 在从连接池获取连接时，是否检查连接的有效性
        dataSource.setValidationQuery("SELECT 1"); // 用于验证连接有效性的查询
    }
    
    // 获取数据库连接
    public static Connection getConnection() throws SQLException {
        return dataSource.getConnection();
    }
    
    // 关闭数据源，释放资源
    public static void close() {
        try {
            dataSource.close(); // 关闭数据源
        } catch (Exception e) {
            e.printStackTrace(); // 打印异常信息
        }
    }

}

二、遍历课表

```
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

public class CourseLister {

    // 列出所有课程
    public static void listCourses() {
        try (Connection conn = DruidDBConnection.getConnection()) { // 获取数据库连接
            // SQL 查询语句，选择课程ID和课程名称
            String courseQuery = "SELECT cid, name FROM course";
            try (PreparedStatement coursePs = conn.prepareStatement(courseQuery); // 准备 SQL 语句
                 ResultSet courseRs = coursePs.executeQuery()) { // 执行查询并返回结果集
    
                System.out.println("课程表：");
                // 遍历结果集
                while (courseRs.next()) {
                    String cid = courseRs.getString("cid"); // 获取课程ID
                    String courseName = courseRs.getString("name"); // 获取课程名称
                    // 打印课程信息
                    System.out.println("课程ID: " + cid + ", 课程名称: " + courseName);
                }
            }
    
        } catch (SQLException e) {
            e.printStackTrace(); // 捕获并打印 SQL 异常
        }
    }
    
    // 主方法，程序入口
    public static void main(String[] args) {
        listCourses(); // 调用列出课程的方法
    }

}

三、查找所有男同学各门作业的平均成绩

```
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

public class StudentCourseAverage {
    public static void main(String[] args) {
        Connection conn = null; // 数据库连接
        PreparedStatement ps = null; // 用于执行 SQL 语句的预编译对象
        ResultSet rs = null; // 存放查询结果的结果集

        try {
            conn = DruidDBConnection.getConnection(); // 获取数据库连接
    
            // 查询课程表
            String courseSql = "SELECT * FROM course";
            ps = conn.prepareStatement(courseSql); // 准备 SQL 查询语句
            rs = ps.executeQuery(); // 执行查询，返回结果集
            while (rs.next()) {
                String courseId = rs.getString("cid"); // 获取课程ID
                String courseName = rs.getString("name"); // 获取课程名称
    
                // 计算男同学作业的平均成绩
                String averageScoreSql = "SELECT AVG(h.grade1) AS h1_avg, AVG(h.grade2) AS h2_avg, AVG(h.grade3) AS h3_avg " +
                        "FROM homework h " +
                        "JOIN student s ON h.sid = s.sid " + // 连接作业表和学生表
                        "WHERE s.gender = N'男' AND h.cid = ?"; // 筛选男同学并指定课程ID
                try (PreparedStatement avgPs = conn.prepareStatement(averageScoreSql)) {
                    avgPs.setString(1, courseId); // 设置课程ID参数
                    try (ResultSet avgRs = avgPs.executeQuery()) { // 执行平均成绩查询
                        if (avgRs.next()) { // 如果结果集有值
                            // 打印课程名称和作业平均成绩
                            System.out.println("课程: " + courseName);
                            System.out.println("作业1平均成绩: " + avgRs.getDouble("h1_avg"));
                            System.out.println("作业2平均成绩: " + avgRs.getDouble("h2_avg"));
                            System.out.println("作业3平均成绩: " + avgRs.getDouble("h3_avg"));
                        }
                    }
                }
            }
        } catch (SQLException e) {
            e.printStackTrace(); // 捕获并打印 SQL 异常
        } finally {
            try {
                // 关闭资源
                if (rs != null) rs.close(); // 关闭结果集
                if (ps != null) ps.close(); // 关闭预编译语句
                if (conn != null) conn.close(); // 关闭数据库连接
            } catch (SQLException e) {
                e.printStackTrace(); // 捕获并打印关闭资源时的异常
            }
        }
    }

}

连接池的优势。

​	连接池的优势在于它可以有效地管理数据库连接的创建和释放，从而提高应用程序的性能和响应速度。通过重用已有的连接，连接池减少了频繁创建和关闭连接的开销，降低了系统资源的消耗，同时还能控制最大连接数，防止数据库过载。此外，连接池通常提供了连接有效性检查和事务管理，增强了数据库操作的可靠性和稳定性。
