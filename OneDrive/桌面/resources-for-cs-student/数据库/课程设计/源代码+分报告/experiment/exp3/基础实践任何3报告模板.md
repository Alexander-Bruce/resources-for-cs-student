// student实体类
public class Student {
    private String sid;           // 学号
    private String name;          // 姓名
    private String gender;        // 性别
    private String majorClass;    // 专业班级
    private Date birth;           // 出生日期
    private String phone;         // 电话号码

    public Student() {}

    public Student(String sid, String name, String gender, String majorClass, String birth, String phone) {
        this.sid = sid;
        this.name = name;
        this.gender = gender;
        this.majorClass = majorClass;
        this.birth = java.sql.Date.valueOf(birth);
        this.phone = phone;
    }

    // Getter和Setter方法
    public String getSid() { return sid; }
    public void setSid(String sid) { this.sid = sid; }

    public String getName() { return name; }
    public void setName(String name) { this.name = name; }

    public String getGender() { return gender; }
    public void setGender(String gender) { this.gender = gender; }

    public String getMajorClass() { return majorClass; }
    public void setMajorClass(String majorClass) { this.majorClass = majorClass; }

    public Date getBirth() { return birth; }
    public void setBirth(Date birth) { this.birth = birth; }

    public String getPhone() { return phone; }
    public void setPhone(String phone) { this.phone = phone; }

    @Override
    public String toString() {
        return "Student{" +
                "sid='" + sid + '\'' +
                ", name='" + name + '\'' +
                ", gender='" + gender + '\'' +
                ", majorClass='" + majorClass + '\'' +
                ", birth=" + birth +
                ", phone='" + phone + '\'' +
                '}';
    }
}

// student接口实现类
public class StudentDaoImpl implements StudentDao {

    // 添加学生
    @Override
    public void addStudent(Student student) {
        // SQL 插入语句
        String sql = "INSERT INTO student (sid, name, gender, major_class, birth, phone) VALUES (?, ?, ?, ?, ?, ?)";
        try (Connection conn = DruidDBConnection.getConnection(); // 获取数据库连接
             PreparedStatement ps = conn.prepareStatement(sql)) { // 创建 PreparedStatement
            ps.setString(1, student.getSid()); // 设置学号
            ps.setString(2, student.getName()); // 设置姓名
            ps.setString(3, student.getGender()); // 设置性别
            ps.setString(4, student.getMajorClass()); // 设置专业班级
            ps.setDate(5, new java.sql.Date(student.getBirth().getTime())); // 设置出生日期
            ps.setString(6, student.getPhone()); // 设置电话号码
            ps.executeUpdate(); // 执行插入操作
        } catch (SQLException e) {
            e.printStackTrace(); // 打印 SQL 异常
        }
    }
    
    // 更新学生信息
    @Override
    public void updateStudent(Student student) {
        // SQL 更新语句
        String sql = "UPDATE student SET name = ?, gender = ?, major_class = ?, birth = ?, phone = ? WHERE sid = ?";
        try (Connection conn = DruidDBConnection.getConnection(); // 获取数据库连接
             PreparedStatement ps = conn.prepareStatement(sql)) { // 创建 PreparedStatement
            ps.setString(1, student.getName()); // 设置姓名
            ps.setString(2, student.getGender()); // 设置性别
            ps.setString(3, student.getMajorClass()); // 设置专业班级
            ps.setDate(4, new java.sql.Date(student.getBirth().getTime())); // 设置出生日期
            ps.setString(5, student.getPhone()); // 设置电话号码
            ps.setString(6, student.getSid()); // 设置学号（条件）
            ps.executeUpdate(); // 执行更新操作
        } catch (SQLException e) {
            e.printStackTrace(); // 打印 SQL 异常
        }
    }
    
    // 删除学生
    @Override
    public void deleteStudent(String sid) {
        // SQL 删除语句
        String sql = "DELETE FROM student WHERE sid = ?";
        try (Connection conn = DruidDBConnection.getConnection(); // 获取数据库连接
             PreparedStatement ps = conn.prepareStatement(sql)) { // 创建 PreparedStatement
            ps.setString(1, sid); // 设置学号（条件）
            ps.executeUpdate(); // 执行删除操作
        } catch (SQLException e) {
            e.printStackTrace(); // 打印 SQL 异常
        }
    }
    
    // 获取指定学号的学生
    @Override
    public Student getStudent(String sid) {
        // SQL 查询语句
        String sql = "SELECT * FROM student WHERE sid = ?";
        Student student = null; // 存储学生对象
        try (Connection conn = DruidDBConnection.getConnection(); // 获取数据库连接
             PreparedStatement ps = conn.prepareStatement(sql)) { // 创建 PreparedStatement
            ps.setString(1, sid); // 设置学号（条件）
            try (ResultSet rs = ps.executeQuery()) { // 执行查询操作
                if (rs.next()) { // 如果有结果
                    student = new Student(); // 创建学生对象
                    student.setSid(rs.getString("sid")); // 设置学号
                    student.setName(rs.getString("name")); // 设置姓名
                    student.setGender(rs.getString("gender")); // 设置性别
                    student.setMajorClass(rs.getString("major_class")); // 设置专业班级
                    student.setBirth(rs.getDate("birth")); // 设置出生日期
                    student.setPhone(rs.getString("phone")); // 设置电话号码
                }
            }
        } catch (SQLException e) {
            e.printStackTrace(); // 打印 SQL 异常
        }
        return student; // 返回学生对象
    }
    
    // 获取所有学生
    @Override
    public List<Student> getAllStudents() {
        // SQL 查询语句
        String sql = "SELECT * FROM student";
        List<Student> students = new ArrayList<>(); // 存储学生列表
        try (Connection conn = DruidDBConnection.getConnection(); // 获取数据库连接
             PreparedStatement ps = conn.prepareStatement(sql); // 创建 PreparedStatement
             ResultSet rs = ps.executeQuery()) { // 执行查询操作
            while (rs.next()) { // 遍历结果集
                Student student = new Student(); // 创建学生对象
                student.setSid(rs.getString("sid")); // 设置学号
                student.setName(rs.getString("name")); // 设置姓名
                student.setGender(rs.getString("gender")); // 设置性别
                student.setMajorClass(rs.getString("major_class")); // 设置专业班级
                student.setBirth(rs.getDate("birth")); // 设置出生日期
                student.setPhone(rs.getString("phone")); // 设置电话号码
                students.add(student); // 添加到学生列表
            }
        } catch (SQLException e) {
            e.printStackTrace(); // 打印 SQL 异常
        }
        return students; // 返回学生列表
    }

}

// homework接口实现类
public class HomeworkDaoImpl implements HomeworkDao {

    /**
     * 添加作业信息
     *
     * @param homework 作业对象，包含作业的详细信息
     * @throws SQLException 如果数据库访问出现错误
     */
    @Override
    public void addHomework(Homework homework) throws SQLException {
        String sql = "INSERT INTO homework (cid, sid, grade1, grade2, grade3) VALUES (?, ?, ?, ?, ?)";
        try (Connection conn = DruidDBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {
            ps.setString(1, homework.getCid());
            ps.setString(2, homework.getSid());
            ps.setBigDecimal(3, toNumeric(homework.getGrade1())); // 转换为 BigDecimal
            ps.setBigDecimal(4, toNumeric(homework.getGrade2())); // 转换为 BigDecimal
            ps.setBigDecimal(5, toNumeric(homework.getGrade3())); // 转换为 BigDecimal
            ps.executeUpdate(); // 执行插入操作
        }
    }
    
    /**
     * 更新已有的作业信息
     *
     * @param homework 作业对象，包含更新后的作业详细信息
     * @throws SQLException 如果数据库访问出现错误
     */
    @Override
    public void updateHomework(Homework homework) throws SQLException {
        String sql = "UPDATE homework SET cid = ?, sid = ?, grade1 = ?, grade2 = ?, grade3 = ? WHERE cid = ? and sid = ?";
        try (Connection conn = DruidDBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {
            ps.setString(1, homework.getCid());
            ps.setString(2, homework.getSid());
            ps.setDouble(3, homework.getGrade1()); // 直接使用 double
            ps.setDouble(4, homework.getGrade2()); // 直接使用 double
            ps.setDouble(5, homework.getGrade3()); // 直接使用 double
            ps.setString(6, homework.getCid());
            ps.setString(7, homework.getSid());
            ps.executeUpdate(); // 执行更新操作
        }
    }
    
    /**
     * 删除指定的作业
     *
     * @param cid 课程ID，指定作业所属课程
     * @param sid 学生ID，指定作业所属学生
     * @throws SQLException 如果数据库访问出现错误
     */
    @Override
    public void deleteHomework(String cid, String sid) throws SQLException {
        String sql = "DELETE FROM homework WHERE cid = ? and sid = ?";
        try (Connection conn = DruidDBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {
            ps.setString(1, cid);
            ps.setString(2, sid);
            ps.executeUpdate(); // 执行删除操作
        }
    }
    
    /**
     * 根据课程ID和学生ID获取作业信息
     *
     * @param cid 课程ID
     * @param sid 学生ID
     * @return 作业对象，包含作业的详细信息，如果没有找到则返回 null
     * @throws SQLException 如果数据库访问出现错误
     */
    @Override
    public Homework getHomeworkById(String cid, String sid) throws SQLException {
        Homework homework = null;
        String sql = "SELECT * FROM homework WHERE cid = ? and sid = ?";
        try (Connection conn = DruidDBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {
            ps.setString(1, cid);
            ps.setString(2, sid);
            try (ResultSet rs = ps.executeQuery()) {
                if (rs.next()) {
                    homework = new Homework(
                            rs.getString("cid"),
                            rs.getString("sid"),
                            rs.getDouble("grade1"),
                            rs.getDouble("grade2"),
                            rs.getDouble("grade3")
                    ); // 创建作业对象
                }
            }
        }
        return homework; // 返回作业对象
    }
    
    /**
     * 获取所有作业信息
     *
     * @return 作业对象的列表，包含所有作业的详细信息
     * @throws SQLException 如果数据库访问出现错误
     */
    @Override
    public List<Homework> getAllHomeworks() {
        List<Homework> homeworks = new ArrayList<>();
        String sql = "SELECT * FROM homework";
        try (Connection conn = DruidDBConnection.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql);
             ResultSet rs = ps.executeQuery()) {
            while (rs.next()) {
                Homework homework = new Homework(
                        rs.getString("cid"),
                        rs.getString("sid"),
                        rs.getDouble("grade1"),
                        rs.getDouble("grade2"),
                        rs.getDouble("grade3")
                ); // 创建作业对象并添加到列表
                homeworks.add(homework);
            }
        } catch (SQLException e) {
            e.printStackTrace(); // 打印异常
        }
        return homeworks; // 返回作业列表
    }
    
    /**
     * 将 double 转换为 BigDecimal，确保在 numeric(4,2) 范围内
     *
     * @param value 输入的 double 值
     * @return 转换后的 BigDecimal 值
     */
    private BigDecimal toNumeric(double value) {
        // 确保范围在 -99.99 到 99.99 之间
        if (value < 0 || value > 200.00) {
            throw new IllegalArgumentException("Value out of range for numeric(4,2): " + value);
        }
        // 转换为 BigDecimal，并设置小数位
        return BigDecimal.valueOf(value).setScale(2, BigDecimal.ROUND_HALF_UP);
    }

}

### 【任务总结】

1. **DAO模型中关键组成要素。**

   DAO（数据访问对象）模型的关键组成要素包括以下几个方面：

   1. **数据源**：
      - DAO 通常与具体的数据源（如关系型数据库、NoSQL 数据库、Web 服务等）进行交互。它负责管理数据的持久化和检索。

   2. **数据模型**：
      - 数据模型表示数据库中的数据结构，通常以类的形式存在。每个数据模型类通常对应数据库中的一张表。

   3. **DAO接口**：
      - DAO 接口定义了数据访问的基本操作，如添加、删除、更新和查询等。这些方法通常是面向对象设计的核心，确保不同实现可以替换而不影响调用者。

   4. **实现类**：
      - DAO 接口的具体实现类实现了接口中定义的方法，并包含与数据源交互的具体逻辑。这些类负责执行 SQL 查询、处理结果集等。

   5. **异常处理**：
      - DAO 层需要处理可能出现的各种异常，并将这些异常转换为应用程序可以理解的形式。

   6. **连接管理**：
      - DAO 负责管理数据库连接，通常通过连接池实现，以提高性能和资源利用率。

   7. **事务管理**：
      - 在涉及多个数据操作的场景中，DAO 需要管理数据库事务，确保数据的一致性和完整性。

   8. **依赖注入**：
      - 在某些设计中，DAO 的实现可能通过依赖注入的方式提供给应用程序，以便于测试和维护。

   9. **映射**：
      - DAO 可与 ORM框架结合使用，简化数据与对象之间的转换。

   这些组成要素共同构成了 DAO 模型，使得数据访问变得更加清晰、可维护和可扩展。

2. **DAO模型中的设计模式。**

   DAO（数据访问对象）模型中常用的设计模式包括以下几种：

   1. **单例模式：
      - 确保 DAO 类只有一个实例，并提供一个全局访问点。这对于管理数据库连接和确保资源共享非常有用。

   2. **工厂模式**：
      - 用于创建 DAO 实例，允许将 DAO 的具体实现与客户端代码分离。工厂模式使得可以在运行时选择不同的 DAO 实现，增强了灵活性。

   3. **策略模式**：
      - 可以定义一系列的数据访问策略，通过不同的策略实现不同的数据库操作。这使得在需要变更数据访问逻辑时更为灵活。

   4. **模板方法模式**：
      - 在 DAO 类中定义一个操作的骨架，并允许子类实现某些步骤。这适用于那些操作过程相似但具体实现不同的 DAO 方法。

   5. **观察者模式：
      - 在某些情况下，DAO 可以与观察者模式结合，以实现数据更改时自动通知相关组件。这可以用于实现数据的实时更新。

   6. **适配器模式**：
      - 用于将不同的数据源（如不同类型的数据库或 API）适配到统一的 DAO 接口，使得上层代码能够无缝使用不同的数据源。

   7. **代理模式**：
      - 通过代理对象控制对 DAO 对象的访问，可以在调用真实 DAO 之前添加额外的功能，如权限检查、日志记录等。

   8. **装饰者模式**：
      - 允许在运行时动态地为 DAO 添加额外的行为，而不影响原始 DAO 的结构。这对于增强功能非常有用。

   9. **适配器模式：
      - 用于将不同的数据源适配到统一的 DAO 接口，使得上层代码能够无缝使用不同的数据源。

   这些设计模式能够帮助开发者构建更为灵活、可维护的 DAO 模型，使得数据访问层能够更好地应对变化和扩展。
