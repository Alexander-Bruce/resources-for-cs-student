#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// 动态规划函数，用于找出最大递增子序列的和
int findMaxValue(vector<int>& dp, const vector<int>& numbers) {
    int maxVal = 0;  // 最大值初始化为0
    for (int i = 0; i < numbers.size(); ++i) {
        dp[i] = numbers[i];  // 初始化dp数组，初始时假设每个元素自成一个序列
        for (int j = 0; j < i; ++j) {  // 遍历当前元素前面的所有元素
            if (numbers[j] < numbers[i]) {  // 判断形成递增序列的条件
                dp[i] = max(dp[i], dp[j] + numbers[i]);  // 更新dp[i]，尝试加上当前元素形成的新序列和
            }
        }
        maxVal = max(maxVal, dp[i]);  // 更新所有情况中的最大值
    }
    return maxVal;  // 返回最大递增子序列的和
}
 
int main() {
    int length;  // 序列的长度
    while (cin >> length) {  // 处理多组输入
        vector<int> numbers(length + 1);  // 存储输入的序列
        vector<int> dp(length + 1, 0);  // dp数组，用于存储到当前位置的最大递增子序列和
        
        for (int i = 1; i <= length; ++i) {  // 读取输入的每个元素
            cin >> numbers[i];
        }
        
        int maxVal = findMaxValue(dp, numbers);  // 计算最大递增子序列的和
        cout << maxVal << endl;  // 输出结果
    }
    return 0;
}
