#include<cstdio>
#include<algorithm>
using namespace std;

struct Contest{
	int begin;
	int end;
	int seconds;
};

Contest contests[1000];
Contest results[1000];

bool compare(Contest a, Contest b){
	return a.seconds < b.seconds;
}

int main() {

	int N;
	while (scanf("%d", &N) != EOF){
		for (int i = 0; i < N; i++){
			scanf("%d%d", &contests[i].begin, &contests[i].end);
			contests[i].seconds = contests[i].end - contests[i].begin;
		}
		sort(contests, contests + N, compare);
		int number = 1;
		results[0] = contests[0];
		for (int i = 1; i < N; i++){
			bool flag = true;
			for (int j = 0; j < number; j++)
				if ((contests[i].begin >= results[j].begin && contests[i].begin < results[j].end) || (contests[i].end > results[j].begin && contests[i].end <= results[j].end))
					flag = false;
			if (flag)
				results[number++] = contests[i];
		}
		printf("%d\n", number);
	}


	return 0;
}
