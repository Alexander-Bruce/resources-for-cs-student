#include<iostream>
#include<cmath>
#include<string>
using namespace std;
int main()
{
	int count;
	string input;
	cin >> count;
	for (int i = 0; i < count; i++)
	{
		cin >> input;
		for (int i = 0; i < input.length(); i++)
		{
			//相邻去重
			if (input[i] == input[i + 1])
			{
				input.erase(i, 1);
				i = -1;
			}
			//12221133345678899
			//1 2
			//2 2
		}	
		cout<<input<<endl;
	}	
 
	return 0;
}
