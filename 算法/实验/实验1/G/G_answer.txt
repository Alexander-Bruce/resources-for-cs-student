bool JudgeIndexSeven(long long int Number)
{
	string NumberString = to_string(Number);
	for (int j = 0; j < NumberString.length(); j++) {
		if (NumberString[j] == '7') 
			return true;
	}
	if (Number % 7 == 0) return true;//pass
	return false;//no pass
}
