void DFS(char** Tarray, int input, bool** visited, int& count, int i, int j) {
	if (visited[i][j] != true) {
		visited[i][j] = true;
		Tarray[i][j] = 'x';
		count++;
		if (i - 1 >= 0 && Tarray[i - 1][j] == 'a' && visited[i - 1][j] == false)
			DFS(Tarray, input, visited, count, i - 1, j);
		if (i + 1 < input && Tarray[i + 1][j] == 'a' && visited[i + 1][j] == false)
			DFS(Tarray, input, visited, count, i + 1, j);
		if (j - 1 >= 0 && Tarray[i][j - 1] == 'a' && visited[i][j - 1] == false)
			DFS(Tarray, input, visited, count, i, j - 1);
		if (j + 1 < input && Tarray[i][j + 1] == 'a' && visited[i][j + 1] == false)
			DFS(Tarray, input, visited, count, i, j + 1);
		//cout<<i<<" "<<j<<endl;
	}
}
int FindMaxSea(char** Tarray,int input) {
	int count = 0;
	int max = 0;
	bool **visited = new bool *[1000];
	for (int i = 0; i < 1000; i++)
		visited[i] = new bool[1000];
	for(int i = 0; i< input; i++)
		for(int j = 0; j< input; j++)
			visited[i][j] = false;
	for (int i = 0; i < input; i++)
		for (int j = 0; j < input; j++)
			if (Tarray[i][j] == 'a' && visited[i][j] == false) {
				count = 0;
				DFS(Tarray, input, visited, count, i, j);
				if (count > max)
					max = count;
			}
	return max;
}
