	#include <iostream>
#include <queue>
#include <vector>
#include <unordered_set>
#include <string>
using namespace std;
const int dx[4] = { 0, -1, 0, 1 }, dy[4] = { -1, 0, 1, 0 };
 
string st, ed;
 
void print(vector<string>& t)
{
    for (int i = 0; i < t.size(); ++i) {
        for (int j = 0; j < 9; ++j) {
            if(t[i][j] == ' ') cout << 0 <<" ";
			else cout << t[i][j] << " ";
        }
        puts("");
    }
}
 
int main()
{
    unordered_set<string> set; 
    queue<vector<string>> q;    
    string temp;
    getline(cin, st);
    for (int i = 0; i < st.size(); i++) {
        if (st[i] == ' ') {
            st.erase(i, 1);
            i--;
        }
	}
    for (int i = 0; i < st.size(); i++)
    {
        if (st[i] == '0')
        {
            st[i] = ' ';
            break;
        }
    }
    getline(cin, ed);
    for (int i = 0; i < ed.size(); i++) {
        if (ed[i] == ' ') {
            ed.erase(i, 1);
            i--;
        }
    }
    for (int i = 0; i < ed.size(); i++)
    {
        if (ed[i] == '0')
        {
            ed[i] = ' ';
            break;
        }
    }
    q.push({ st });
    set.insert(st);
    int times = 0;   
    int flag = 0;
    while (!q.empty()) {
        auto t = q.front(); q.pop();
        if (t.back() == ed) {  
            if (t.size() != 0) {
                cout << t.size() - 1 << endl;
                print(t);
                goto a;
            }
        }
        int x1 = t.back().find(' ');
        int y1 = x1 % 3;
        x1 /= 3;
 
        ++times;
 
        for (int k = 0; k < 4; k++) {
            int x = dx[k] + x1, y = dy[k] + y1;
            if (x < 0 || y < 0 || x > 2 || y > 2) continue;
 
            string tmp = t.back();
            swap(tmp[x1 * 3 + y1], tmp[x * 3 + y]);
            if (set.count(tmp)) continue;
            set.insert(tmp);
            t.emplace_back(tmp);
            q.emplace(t);
 
            t.pop_back();
        }
    }
    if (flag == 0)
        cout << "-1" << endl;
    a:return 0;
}
