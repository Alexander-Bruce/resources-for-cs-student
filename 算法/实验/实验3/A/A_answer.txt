#include <iostream>
#include <vector>
 
using namespace std;
 
long long int binarySearch(vector<long long int>& assignedNumber, long long int toldNumber, long long int left, long long int right) {
    long long int mid = (left + right) / 2;
    while (left <= right) {
        if (assignedNumber[mid] == toldNumber) {
            return mid + 1;
        }
        else if (assignedNumber[mid] < toldNumber) {
            left = mid + 1;
        }
        else {
            right = mid - 1;
        }
        mid = (left + right) / 2;
    }
    return left;
}
 
int main() {
    long long int round, member;
    while (cin >> member >> round) {
        vector<long long int> assignedNumber(member);
        vector<long long int> toldNumber(round);
        vector<long long int> result;
 
        for (long long int i = 0; i < member; ++i) {
            cin >> assignedNumber[i];
        }
 
        for (long long int i = 0; i < round; ++i) {
            cin >> toldNumber[i];
        }
 
        for (long long int i = 0; i < round; ++i) {
            long long int told = toldNumber[i];
            long long int resultIndex = binarySearch(assignedNumber, told, 0, assignedNumber.size() - 1);
            if (resultIndex != 0) {
                result.push_back(assignedNumber[resultIndex - 1]);
            }
            else {
                result.push_back(assignedNumber[0]);
            }
            //cout << told << " " << resultIndex << endl;
        }
 
        for (long long int i = 0; i < result.size(); ++i) {
            if (i == result.size() - 1)
                cout << result[i];
            else cout << result[i] << " ";
        }
        cout << endl;
    }
 
    return 0;
}
