#include <iostream>
#include <vector>
#include <math.h>
#include <algorithm>
using namespace std;

int level;  // 存储三角形的层数
vector<vector<int>> value;  // 用来存储三角形中每层的值的二维向量

// 动态规划函数，计算从顶部到底部的最大路径总和
void FindMaxValueTreasure(int row, int n) {
    for (int i = level - 1; i >= 1; i--)  // 从倒数第二层开始向上计算，直到顶层
    {
        for (int k = 1; k <= i; k++)  // 遍历当前层的每一个数字
        {
            // 更新当前位置的最大总和，即当前位置的值加上下一层相邻两个位置的较大值
            value[i][k] += max(value[i + 1][k], value[i + 1][k + 1]);
        }
    }
}

int main() {
    int times;  // 测试用例的数量
    cin >> times;  // 输入测试用例的数量
    for (int t = 0; t < times; t++) {  // 遍历每一个测试用例
        cin >> level;  // 输入三角形的层数
        value.resize(level + 1, vector<int>(level + 1));  // 重新设置value的大小，每一层的长度和层数相同
        for (int row = 1; row <= level; row++) {  // 输入每层的值
            for (int n = 1; n <= row; n++) {
                cin >> value[row][n];
            }
        }
        FindMaxValueTreasure(level, level);  // 调用函数计算最大路径总和
        if (t != times - 1)
            cout << value[1][1] << endl;  // 输出结果，并在非最后一次测试用例后输出换行符
        else
            cout << value[1][1];  // 最后一次输出后不加换行符
    }
    return 0;
}
