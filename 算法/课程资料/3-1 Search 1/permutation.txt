// permutation
#include<bits/stdc++.h>
using namespace std;
int n;
int num[10], vis[10];

void dfs(int step){
	if(step==n+1){
		for(int i=1;i<=n;i++)
			printf("%d ",num[i]);
		printf("\n");
		return;
	}
	for(int i=1;i<=n;i++){
		if(vis[i]==0){
			num[step]=i;
			vis[i]=1;
			dfs(step+1);
			vis[i]=0; // backtracking
		}
	}
}

int main(){
	scanf("%d",&n);
	memset(vis,0,sizeof(vis));
	dfs(1);
	return 0;
}
