#include<bits/stdc++.h>
using namespace std;
int n,c,x[100005]; // n, c and coordinates 

bool check(int dis){
    int cnt=1, place=0; // the first cow，placed at the first compartment
    for (int i = 1; i < n; ++i)  // check the compartments one by one
        if (x[i] - x[place] >= dis){   
            cnt++;     // place a cow
            place = i; // update the coordinate
        }
    if (cnt >= c) return true;  // there are enough compartments
    else          return false; // Not enough compartments
}

int main(){
    scanf("%d%d",&n, &c);
    for(int i=0;i<n;i++)    scanf("%d",&x[i]);
    sort(x,x+n);                    // sort coordinates
    int left=0, right=x[n-1]-x[0];
    int ans = 0;
    while(left < right){
        int mid = left + (right - left)/2;     // binary search
        // Check whether there are enough compartments when the minimum distance is mid
        if(check(mid)){
            ans = mid;              // Yes, save mid
            left = mid + 1;         // increase the distance
           }
        else    right = mid;         // No, reduce the distance
    }
    cout << ans; // print the answer
    return 0;
}
